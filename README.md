# node-red-standalone

- npm install -g windows-build-tools
  (Run the script from an administrative PowerShell)
- git clone https://github.com/kazuhitoyokoi/node-red-standalone.git
- cd node-red-standalone
- npm install
- npm run build
- npm start
- npm start -- --devepment
- npm start -- --production
- npm start -- --dashboard
- npm start -- --worldmap
